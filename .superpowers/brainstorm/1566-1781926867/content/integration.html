egui 预览中的 SVG 渲染

egui 原生不支持 SVG。需要将 SVG 转为光栅图像：

// 方案：使用 resvg + tiny-skia 将 SVG 渲染为 egui::ColorImage
fn render_svg_to_egui_image(svg: &str, scale: f32) -> egui::ColorImage {
    let tree = resvg::usvg::Tree::from_str(svg, &resvg::usvg::Options::default())?;
    let size = tree.size();
    let mut pixmap = resvg::tiny_skia::Pixmap::new(
        (size.width() * scale) as u32,
        (size.height() * scale) as u32,
    )?;
    resvg::render(&tree, tiny_skia::Transform::from_scale(scale, scale), &mut pixmap);
    // pixmap → egui::ColorImage
}

需要新增依赖：resvg = "0.42", tiny-skia = "0.11"
